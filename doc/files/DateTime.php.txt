class DateTime extends \DateTime
{
    private static $_sFormat = 'Y-m-d';

    public function __toString()
    {
        return $this->format(self::$_sFormat);
    }

    public static function setFormat($s)
    {
        self::$_sFormat = $s;
    }
}
